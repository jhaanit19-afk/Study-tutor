async function uploadNotes() { const file = document.getElementById('noteFile').files[0]; const title = document.getElementById('noteTitle').value; if (!file || !title) { alert("Please enter title and select a file!"); return; } try { // 1️⃣ Upload file to Firebase Storage const storageRef = storage.ref('notes/' + file.name); await storageRef.put(file); // 2️⃣ Get file URL const url = await storageRef.getDownloadURL(); // 3️⃣ Save note info in Firestore await db.collection('notes').add({ title: title, fileUrl: url, uploadedBy: 'Tutor1', date: new Date() }); alert('Note uploaded successfully!'); } catch (error) { console.error(error); alert('Upload failed!'); } }
